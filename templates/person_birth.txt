SELECT ?x ?type (group_concat(distinct ?person_birthday;separator=",") as ?person_birthdays) (group_concat(distinct ?person_mother;separator=",") as ?person_mothers) (group_concat(distinct ?person_father;separator=",") as ?person_fathers) (group_concat(distinct ?person_birthplace;separator=",") as ?person_birthplaces)

OPTIONAL {
	?x <http://erlangen-crm.org/170309/P92i_was_brought_into_existence_by> ?y1 .

OPTIONAL {
	?x <http://erlangen-crm.org/170309/P92i_was_brought_into_existence_by> ?y3 .

	?y3 <http://erlangen-crm.org/170309/P96_by_mother> ?person_mother .
	?person_mother rdf:type <http://visit.de/ontologies/vismo/Person> . }
OPTIONAL {
	?x <http://erlangen-crm.org/170309/P92i_was_brought_into_existence_by> ?y4 .
	?y4 rdf:type <http://erlangen-crm.org/170309/E67_Birth> .

	?x <http://erlangen-crm.org/170309/P92i_was_brought_into_existence_by> ?y5 .

	?person_birthplace rdf:type <http://visit.de/ontologies/vismo/Place> . }

FILTER regex(str(?x), "^ADD_ID_HERE$", "")}
